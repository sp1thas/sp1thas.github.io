def parse_list_fields(x):
    if isinstance(x, str):
        return json.loads(x)
    elif isinstance(x, list):
        return x
    else:
        return []

df['categories'] = df['categories'].apply(parse_list_fields)
df['authors'] = df['authors'].apply(parse_list_fields)
